# projetos-de-educa-o-digital import uuid import datetime class CertificadoSustentavel: def __init__(self, propriedade, hectares_app, hectares_producao): self.id_rastreio = str(uuid.uuid4())[:8].upper() self.propriedade = propriedade self.hectares_app = hectares_app self.hectares_producao = hectares_producao self.data_emissao = datetime.date.today() self.status_sustentabilidade = "Em análise" def verificar_conformidade(self, car_ativo, adota_bioinsumos, emissao_zero): # Regra de negócio simples para classificar o nível sustentável if car_ativo and adota_bioinsumos and emissao_zero: self.status_sustentabilidade = "OURO (Sustentabilidade Avançada)" elif car_ativo and (adota_bioinsumos or emissao_zero): self.status_sustentabilidade = "PRATA (Transição Ecológica)" else: self.status_sustentabilidade = "PENDENTE (Requer adequação)" return self.status_sustentabilidade def exibir_codigo(self): print(f"--- CÓDIGO DE RASTREIO DO AGRO ---") print(f"Código: AGRO-BR-{self.id_rastreio}") print(f"Propriedade: {self.propriedade}") print(f"Nível: {self.status_sustentabilidade}") print(f"Emissão: {self.data_emissao}") # Exemplo de Uso fazenda = CertificadoSustentavel("Sítio Verde", hectares_app=20, hectares_producao=80) fazenda.verificar_conformidade(car_ativo=True, adota_bioinsumos=True, emissao_zero=True) fazenda.exibir_codigo() class SistemaIrrigacao: def __init__(self, umidade_minima, capacidade_reservatorio): self.umidade_minima = umidade_minima self.reservatorio = capacidade_reservatorio self.ligado = False def monitorar_solo(self, umidade_atual, previsao_chuva): print(f"Umidade do solo: {umidade_atual}% | Previsão de chuva: {previsao_chuva}") if previsao_chuva == "Alta": print("Economia de água ativada. Irrigação suspensa devido à previsão de chuva.") self.desligar_sistema() elif umidade_atual < self.umidade_minima: print("Solo seco detectado. Acionando irrigação...") self.ligar_sistema() self.reservatorio -= 50 # Simulando gasto de água else: print("Umidade ideal. Sistema desligado.") self.desligar_sistema() def ligar_sistema(self): self.ligado = True print("Status: SISTEMA LIGADO.") def desligar_sistema(self): self.ligado = False print("Status: SISTEMA DESLIGADO.") # Exemplo de Uso trator_inteligente = SistemaIrrigacao(umidade_minima=40, capacidade_reservatorio=1000) trator_inteligente.monitorar_solo(umidade_atual=35, previsao_chuva="Baixa")
