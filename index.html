# projetos-de-educa-o-digital import uuid import datetime class CertificadoSustentavel: def __init__(self, propriedade, hectares_app, hectares_producao): self.id_rastreio = str(uuid.uuid4())[:8].upper() self.propriedade = propriedade self.hectares_app = hectares_app self.hectares_producao = hectares_producao self.data_emissao = datetime.date.today() self.status_sustentabilidade = "Em análise" def verificar_conformidade(self, car_ativo, adota_bioinsumos, emissao_zero): # Regra de negócio simples para classificar o nível sustentável if car_ativo and adota_bioinsumos and emissao_zero: self.status_sustentabilidade = "OURO (Sustentabilidade Avançada)" elif car_ativo and (adota_bioinsumos or emissao_zero): self.status_sustentabilidade = "PRATA (Transição Ecológica)" else: self.status_sustentabilidade = "PENDENTE (Requer adequação)" return self.status_sustentabilidade def exibir_codigo(self): print(f"--- CÓDIGO DE RASTREIO DO AGRO ---") print(f"Código: AGRO-BR-{self.id_rastreio}") print(f"Propriedade: {self.propriedade}") print(f"Nível: {self.status_sustentabilidade}") print(f"Emissão: {self.data_emissao}") # Exemplo de Uso fazenda = CertificadoSustentavel("Sítio Verde", hectares_app=20, hectares_producao=80) fazenda.verificar_conformidade(car_ativo=True, adota_bioinsumos=True, emissao_zero=True) fazenda.exibir_codigo()
